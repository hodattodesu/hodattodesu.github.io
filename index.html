raw_data <- raw_data[,c( "Product_Collection", "Vertical_Segment","Recommended_Customer_Price","nb_of_Cores", "Processor_Base_Frequency", "Cache", "TDP", "Max_Memory_Bandwidth", "T","Lithography","nb_of_Threads","Max_nb_of_PCI_Express_Lanes")]

raw_data$"Cache" <- gsub(" SmartCache","",raw_data$"Cache")
raw_data$"Cache" <- gsub(" L2","",raw_data$"Cache")

for (x in 1:nrow(raw_data)){
    
    if (str_detect(raw_data[x,6]," MB"))
        raw_data[x,6] <- gsub(" MB","000",raw_data[x,6])
    else
        raw_data[x,6] <- gsub(" KB","",raw_data[x,6])
    
    if (str_detect(raw_data[x,5]," GHz"))
        raw_data[x,5] <- gsub(" GHz","0",raw_data[x,5])
    else
        raw_data[x,5] <- gsub(" MHz","",raw_data[x,5])
    
    if (str_detect(raw_data[x,8]," GHz"))
        raw_data[x,8] <- gsub(" GHz","0",raw_data[x,8])
    else
        raw_data[x,8] <- gsub(" MHz","",raw_data[x,8])
}

#Bỏ các đơn vị chuỗi
raw_data$"T" <- gsub("°C", "", raw_data$"T", fixed = TRUE)
raw_data$"TDP" <- gsub(" W","",raw_data$"TDP")
raw_data$"Max_Memory_Bandwidth" <- gsub(" GB/s","",raw_data$"Max_Memory_Bandwidth")
raw_data$"Recommended_Customer_Price" <- gsub(".00","",raw_data$"Recommended_Customer_Price")
raw_data$"Recommended_Customer_Price" <- gsub("[^[:alnum:] ]","",raw_data$"Recommended_Customer_Price")
raw_data$"Processor_Base_Frequency" <- gsub("[^[:alnum:] ]","",raw_data$"Processor_Base_Frequency")
raw_data$"Lithography" <- gsub(" nm","",raw_data$"Lithography")

#Tiến hành chuyển ký tự chuỗi trong bảng về lại số
suppressWarnings(raw_data$"Recommended_Customer_Price" <- as.integer(raw_data$"Recommended_Customer_Price"))

suppressWarnings(raw_data$"Cache" <- as.integer(raw_data$"Cache"))

suppressWarnings(raw_data$"TDP" <- as.numeric(raw_data$"TDP"))
